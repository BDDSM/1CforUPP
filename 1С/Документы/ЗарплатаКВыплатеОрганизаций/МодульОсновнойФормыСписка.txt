Процедура УстановитьЗаголовкиОрганизацийВФорме()
	
	ЭлементыФормы.ДокументСписок.Колонки.Организация.ТекстШапки = ОбщегоНазначения.ПреобразоватьСтрокуИнтерфейса("Организация");
	ЭтаФорма.Заголовок = ОбщегоНазначения.ПреобразоватьСтрокуИнтерфейса("Зарплата к выплате организации");
	ЭлементыФормы.ОрганизацияФлаг.Заголовок = ОбщегоНазначения.ПреобразоватьСтрокуИнтерфейса("Организация:");
	
КонецПроцедуры // УстановитьЗаголовкиОрганизацийВФорме()

КонецПроцедуры // ОбработчикОжиданияДокументСписокПриАктивизацииСтроки

////////////////////////////////////////////////////////////////////////////////
// ОБРАБОТЧИКИ СОБЫТИЙ ФОРМЫ

Процедура ПередОткрытием(Отказ, СтандартнаяОбработка)

	ЗарплатаКВыплатеОрганизацийПереопределяемый.ФормаСпискаПередОткрытиемДополнительно(ЭтаФорма);
	
КонецПроцедуры

	УстановитьЗаголовкиОрганизацийВФорме();
	
	РаботаСДиалогамиЗК.НастроитьОтборПоОрганизации(ЭтаФорма, "ДокументСписок");
	
КонецПроцедуры

	ЗарплатаКВыплатеОрганизацийПереопределяемый.ФормаСпискаОбработкаОповещенияДополнительно(ЭтаФорма, ИмяСобытия, Параметр, Источник);
	
КонецПроцедуры

////////////////////////////////////////////////////////////////////////////////
// ПРОЦЕДУРЫ - ДЕЙСТВИЯ КОМАНДНЫХ ПАНЕЛЕЙ ФОРМЫ

Процедура ДействияФормыВыплатить(Кнопка)
	
	ЗарплатаКВыплатеОрганизацийПереопределяемый.ФормаСпискаОбработатьКомандыВыплаты(ЭтаФорма, Кнопка);
	
КонецПроцедуры

// Процедура вызывается при выборе пункта подменю "Движения документа по регистрам" меню "Перейти".
// командной панели формы. Процедура отрабатывает печать движений документа по регистрам.
//
Процедура ДействияФормыДвиженияДокументаПоРегистрам(Кнопка)

	Если ЭлементыФормы.ДокументСписок.ТекущиеДанные = Неопределено Тогда
		Возврат;
	КонецЕсли;

	РаботаСДиалогами.НапечататьДвиженияДокумента(ЭлементыФормы.ДокументСписок.ТекущиеДанные.Ссылка);

КонецПроцедуры // ДействияФормыДвиженияДокументаПоРегистрам()

// Процедура вызывается при выборе пункта подменю "Структура подчиненности" меню "Перейти".
Процедура ДействияФормыСтруктураПодчиненностиДокумента(Кнопка)
	
	Если ЭлементыФормы.ДокументСписок.ТекущиеДанные = Неопределено тогда
		Возврат
	КонецЕсли;
	
	РаботаСДиалогами.ПоказатьСтруктуруПодчиненностиДокумента(ЭлементыФормы.ДокументСписок.ТекущиеДанные.Ссылка);
	
КонецПроцедуры

// Процедура вызывается при нажатии кнопки "Печать" командной панели формы,
// вызывает печать по умолчанию для формы документа.
//
Процедура ДействияФормыДействиеПечать(Кнопка)

	Если ЭлементыФормы.ДокументСписок.ТекущиеДанные <> Неопределено Тогда

		Попытка
			УниверсальныеМеханизмы.НапечататьДокументИзФормыСписка(ЭлементыФормы.ДокументСписок.ТекущиеДанные.Ссылка.ПолучитьОбъект())
		Исключение
		КонецПопытки
			
	КонецЕсли;

КонецПроцедуры // ОсновныеДействияФормыДействиеПечать()

// Процедура обеспечивает синхронизацию строк табличных полей 
//
// Параметры:
//	Элемент - табличное поле, которое отображает т.ч.
//	
Процедура ДокументСписокПриАктивизацииСтроки(Элемент)
    
    ПодключитьОбработчикОжидания("ОбработчикОжиданияДокументСписокПриАктивизацииСтроки", 0.05, Истина);	 	
	
КонецПроцедуры // ДокументСписокПриАктивизацииСтроки()

Процедура ДокументСписокПриПолученииДанных(Элемент, ОформленияСтрок)
	
	Если НЕ ЭлементыФормы.ДокументСписок.Колонки.Оплачена.Видимость Тогда
		Возврат
	КонецЕсли;
	
	Ссылки = Новый Массив();
	Для каждого ОформлениеСтроки Из ОформленияСтрок Цикл
		Ссылки.Добавить(ОформлениеСтроки.ДанныеСтроки.Ссылка);
	КонецЦикла;
	
	СостояниеДокументов = ЗарплатаКВыплатеОрганизацийПереопределяемый.ПолучитьСостояние(Ссылки);

	ЗарплатаКВыплатеОрганизацийСервис.ПоказатьСостояние(СостояниеДокументов, ОформленияСтрок, "Оплачена");	

КонецПроцедуры

Процедура ОрганизацияПриИзменении(Элемент)
	
	РаботаСДиалогамиЗК.ОбработкаИзмененияБыстрогоОтбораПоОрганизации(ЭтаФорма, Элемент, "ДокументСписок");

КонецПроцедуры

Процедура ДокументСписокПередНачаломДобавления(Элемент, Отказ, Копирование)
	
	РаботаСДиалогамиЗК.СписокДокументовПередНачаломДобавления(ЭтаФорма, Отказ, Копирование, Отбор.Организация.Значение, "ЗарплатаКВыплатеОрганизаций");

КонецПроцедуры

ЗарплатаКВыплатеОрганизацийПереопределяемый.ФормаСпискаСоздатьКомандыВыплаты(ЭтаФорма, Новый Действие("ДействияФормыВыплатить"));

ЗарплатаКВыплатеОрганизацийПереопределяемый.ФормаСпискаПриСозданииФормыДополнительно(ЭтаФорма);
