Перем ПредставленияПериодДетализации;   // Коллекция представлений реквизита ПериодДетализации 
                                      	// Используется для вывода представления значений реквизитов в списке.

//ОБРАБОТЧИКИ СОБЫТИЙ ФОРМЫ И ЭЛЕМЕНТОВ ФОРМЫ
 
// Обработчик события ПриПолученииДанных табличного поля СправочникСписок
// Отображает представление реквизита ПериодДетализации
Процедура СправочникСписокПриПолученииДанных(Элемент, ОформленияСтрок)
	
	// Установим представления в колонках ВидРегулярногоДокумента, ПериодДетализации
	Если НЕ ЭлементыФормы.СправочникСписок.Колонки.ПериодДетализации.Видимость 
	   И НЕ ЭлементыФормы.СправочникСписок.Колонки.ВидРегулярногоДокумента.Видимость Тогда
		Возврат;
	КонецЕсли;
	
	СписокВидовДокументов = РегламентноеФормированиеДокументов.РегламентФормированияДокументовВыпуска_СписокВидовРегулярныхДокументов();
	
	Для Каждого ОформлениеСтроки Из ОформленияСтрок Цикл
		
		Если ЭлементыФормы.СправочникСписок.Колонки.ВидРегулярногоДокумента.Видимость Тогда
			Значение = СписокВидовДокументов.НайтиПоЗначению(ОформлениеСтроки.ДанныеСтроки.ВидРегулярногоДокумента);
			Если Значение <> Неопределено Тогда
				ОформлениеСтроки.Ячейки.ВидРегулярногоДокумента.УстановитьТекст(Значение.Представление);
			КонецЕсли;
		КонецЕсли;
		
		Если ЭлементыФормы.СправочникСписок.Колонки.ПериодДетализации.Видимость Тогда
			ОформлениеСтроки.Ячейки.ПериодДетализации.УстановитьТекст(РегламентноеФормированиеДокументов.ПолучитьПредставление(
				ПредставленияПериодДетализации, 
				ОформлениеСтроки.ДанныеСтроки.ПериодДетализации)
			);
		КонецЕсли;
		
	КонецЦикла;
	
КонецПроцедуры

//Описания представлений реквизита "ПериодДетализации" используются также в форме элемента
РегламентноеФормированиеДокументов.ДобавитьЭлементКоллекцииПредставлений(ПредставленияПериодДетализации, Перечисления.Периодичность.День,   "1 документ в день",	1);
РегламентноеФормированиеДокументов.ДобавитьЭлементКоллекцииПредставлений(ПредставленияПериодДетализации, Перечисления.Периодичность.Неделя, "1 документ в неделю",	7);
РегламентноеФормированиеДокументов.ДобавитьЭлементКоллекцииПредставлений(ПредставленияПериодДетализации, Перечисления.Периодичность.Декада, "1 документ в декаду",	10);
РегламентноеФормированиеДокументов.ДобавитьЭлементКоллекцииПредставлений(ПредставленияПериодДетализации, Перечисления.Периодичность.Месяц,  "1 документ в месяц",	30);
