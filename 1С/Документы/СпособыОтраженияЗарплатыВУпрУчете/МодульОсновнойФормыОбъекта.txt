// Назначает доступность элементам управления при изменении данных
//
// Параметры
//  нет
//
Процедура НазначитьДоступность()
	ХарактерЗатрат = СтатьяЗатрат.ХарактерЗатрат;
	ЭлементыФормы.ОбъектСтроительства.Доступность = ХарактерЗатрат = Перечисления.ХарактерЗатрат.ВложенияВоВнеоборотныеАктивы;
	ЭлементыФормы.СпособРаспределенияЗатрат.Доступность = ХарактерЗатрат = Перечисления.ХарактерЗатрат.ПроизводственныеРасходы;
КонецПроцедуры // НазначитьВидимость()

///////////////////////////////////////////////////////////////////////////////
// ОБРАБОТЧИКИ СОБЫТИЙ ФОРМЫ

Процедура ПриОткрытии()
	НазначитьДоступность();
КонецПроцедуры

Процедура ПриЗаписи(Отказ)
	Если ЗначениеЗаполнено(ОбъектСтроительства) И СтатьяЗатрат.ХарактерЗатрат <> Перечисления.ХарактерЗатрат.ВложенияВоВнеоборотныеАктивы Тогда
		Сообщить("Для данного способа отражения зарплаты в учете не нужно задавать объект строительства. Значение очищено.");
		ОбъектСтроительства = Справочники.ОбъектыСтроительства.ПустаяСсылка();
	
	КонецЕсли;
	Если ЗначениеЗаполнено(СпособРаспределенияЗатрат) И СтатьяЗатрат.ХарактерЗатрат <> Перечисления.ХарактерЗатрат.ПроизводственныеРасходы Тогда
		Сообщить("Для данного способа отражения зарплаты в учете не нужно задавать способ распределения затрат. Значение очищено.");
		СпособРаспределенияЗатрат = Справочники.СпособыРаспределенияЗатратНаВыпуск.ПустаяСсылка();
	
	КонецЕсли;
КонецПроцедуры

////////////////////////////////////////////////////////////////////////////////
// ОБРАБОТЧИКИ СОБЫТИЙ ЭЛЕМЕНТОВ ФОРМЫ

Процедура СтатьяЗатратПриИзменении(Элемент)
	НазначитьДоступность();
КонецПроцедуры
