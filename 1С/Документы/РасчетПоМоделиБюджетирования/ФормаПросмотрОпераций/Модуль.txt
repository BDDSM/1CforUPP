Процедура БюджетныеОперацииПриВыводеСтроки(Элемент, ОформлениеСтроки, ДанныеСтроки)
	
	ОформлениеСтроки.Ячейки.Период.Текст=Формат(ДанныеСтроки.Период,ФорматПериод);
	ОформлениеСтроки.Ячейки.Период.ОтображатьТекст=Истина;
	
КонецПроцедуры

Процедура ДействияФормыИзменение(Кнопка)
	
	Если НЕ ЭлементыФормы.ПанельИзменение.Свертка = РежимСверткиЭлементаУправления.Верх Тогда
		ЭлементыФормы.ПанельИзменение.Свертка = РежимСверткиЭлементаУправления.Верх;
	Иначе
		ЭлементыФормы.ПанельИзменение.Свертка = РежимСверткиЭлементаУправления.Нет;
	КонецЕсли;
	
	Если ЭлементыФормы.ПанельИзменение.Свертка = РежимСверткиЭлементаУправления.Верх Тогда
		ЭлементыФормы.ДействияФормы.Кнопки.Изменение.Пометка = Ложь;
	Иначе
		ЭлементыФормы.ДействияФормы.Кнопки.Изменение.Пометка = Истина;
	КонецЕсли;
		
	ЭлементыФормы.БюджетныеОперации.ТолькоПросмотр=НЕ ЭлементыФормы.ДействияФормы.Кнопки.Изменение.Пометка;
	
КонецПроцедуры

Процедура ПриОткрытии()
	
	ЭлементыФормы.ПанельИзменение.Свертка = РежимСверткиЭлементаУправления.Верх;
	ЭлементыФормы.ДействияФормы.Кнопки.Изменение.Пометка = Ложь;
	ЭлементыФормы.БюджетныеОперации.ТолькоПросмотр=Истина;
	
КонецПроцедуры
