Перем СписокТиповДокументовПФР;
Перем СписокТиповДокументовФСС;
Перем СписокТиповДокументовФСГС;

Перем СоответствиеСпискаТипуПолучателя;

Процедура ТипПолучателяПриИзменении(Элемент)
	ОграничитьТипДокумента();
КонецПроцедуры

Процедура ПередОткрытием(Отказ, СтандартнаяОбработка)
	ОграничитьТипДокумента();
КонецПроцедуры

Процедура ОграничитьТипДокумента()
	Если ЗначениеЗаполнено(ТипПолучателя) Тогда
		ЭлементыФормы.ТипДокумента.ДоступныеЗначения = СоответствиеСпискаТипуПолучателя[ТипПолучателя];
		
		//при необходимости - очистим поле ввода
		ТекущееЗначение = ЭлементыФормы.ТипДокумента.Значение;
		Если СоответствиеСпискаТипуПолучателя[ТипПолучателя].НайтиПоЗначению(ТекущееЗначение) = Неопределено Тогда
			ЭлементыФормы.ТипДокумента.Значение = Неопределено;
		КонецЕсли;
	Иначе
		ЭлементыФормы.ТипДокумента.Значение = Неопределено;
		ЭлементыФормы.ТипДокумента.ДоступныеЗначения = Новый СписокЗначений;
	КонецЕсли;
КонецПроцедуры

СписокТиповДокументовФНС = Новый СписокЗначений;
