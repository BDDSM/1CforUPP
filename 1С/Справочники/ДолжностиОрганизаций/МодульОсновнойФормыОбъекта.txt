Процедура ОбработкаОповещения(ИмяСобытия, Параметр, Источник)
	
	Если ИмяСобытия = "ОбновитьФорму" и (Источник = Ссылка или (ТипЗнч(Источник) = Тип("СправочникСсылка.ДолжностиОрганизаций") и Источник.Пустая())) Тогда
		
		ИмяОбновляемогоЭлемента = Параметр.ИмяЭлемента;
		РаботаСФайламиПереопределяемый.ОбновитьПредставлениеЭлемента(ИмяОбновляемогоЭлемента, ЭтаФорма, Ссылка);
		
	КонецЕсли;
	
КонецПроцедуры

Процедура ПередОткрытием(Отказ, СтандартнаяОбработка)
	
	ДолжностиОрганизацийКлиент.ФормаЭлементаПередОткрытием(Отказ, СтандартнаяОбработка, ЭтаФорма);
		
	ДополнительныеДействияПриИзменении = Новый Действие("ДополнительныеДействияПриИзменении");
	
	РаботаСФайламиПереопределяемый.ПередОткрытиемФормыЭлементаДополнительно(ЭтаФорма, Новый Действие("ДополнительныеДействия"), Новый Структура("ПриВыводеСтроки", Новый Действие("ДополнительноеДействиеПриВыводеСтроки")));
	
	ДолжностиОрганизацийДополнительный.ФормаЭлементаПередОткрытиемДополнительно(ЭтаФорма, ДополнительныеДействияПриИзменении);
	
	ИзменениеАктуальностиЭлементовПереопределяемый.ФормаЭлементаПередОткрытиемДополнительно(ЭтаФорма, ДополнительныеДействияПриИзменении);
	
КонецПроцедуры

Процедура ПриОткрытии()
	
	ДолжностиОрганизацийКлиент.ФормаЭлементаПриОткрытии(ЭтаФорма);
	
	УстановитьЗаголовкиОрганизацийВФорме();
	
	ИзменениеАктуальностиЭлементовПереопределяемый.ФормаЭлементаПриОткрытииДополнительно(ЭтаФорма);
	
КонецПроцедуры
