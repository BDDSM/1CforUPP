// Процедура устанавливает видимость / доступность элементов формы
//
Процедура ДействияФормыРедактироватьКод(Кнопка)
	МеханизмНумерацииОбъектов.ИзменениеВозможностиРедактированияНомера(ЭтотОбъект.Метаданные(), ЭтаФорма, ЭлементыФормы.ДействияФормы.Кнопки.Подменю, ЭлементыФормы.Код);
КонецПроцедуры

Процедура УстановитьВидимость()
	
	ПВХ = ПланыВидовХарактеристик.ВидыРезультатовАнализаНоменклатуры;
	Страницы = ЭлементыФормы.ПанельДопустимыеЗначениеПоказателей.Страницы;
	ЭлементыФормы.НадписьТекстПояснение.Заголовок = "";
	
	Если НЕ ЗначениеЗаполнено(ВидРезультатаАнализа) ИЛИ НЕ ВидРезультатаАнализа.Предопределенный Тогда
		
		Страницы["ВхождениеВСписок"].Видимость = Ложь;
		Страницы["ЧисловойДиапазон"].Видимость = Ложь;
		
	ИначеЕсли ВидРезультатаАнализа = ПВХ.ЗначениеИзСписка Тогда
		
		Страницы["ВхождениеВСписок"].Видимость = Истина;
		Страницы["ЧисловойДиапазон"].Видимость = Ложь;
		
		ЭлементыФормы.ПанельДопустимыеЗначениеПоказателей.ТекущаяСтраница = Страницы["ВхождениеВСписок"];
		
		ЭлементыФормы.НадписьТекстПояснение.Заголовок = "Допустимым результатом замеров будет значение из списка.";
		
	ИначеЕсли ВидРезультатаАнализа = ПВХ.ЧислоВИнтервале  Тогда
		
		Страницы["ВхождениеВСписок"].Видимость = Ложь;
		Страницы["ЧисловойДиапазон"].Видимость = Истина;
		
		ЭлементыФормы.НадписьМинЗначение .Заголовок = "Минимальное значение:";
		ЭлементыФормы.НадписьМаксЗначение.Заголовок = "Максимальное значение:";
		
		ЭлементыФормы.ПанельДопустимыеЗначениеПоказателей.ТекущаяСтраница = Страницы["ЧисловойДиапазон"];
		
		ЭлементыФормы.НадписьТекстПояснение.Заголовок = "Допустимым результатом замеров будет число входящее в указанный диапазон.";
		
	ИначеЕсли ВидРезультатаАнализа = ПВХ.Погрешность  Тогда
		
		Страницы["ВхождениеВСписок"].Видимость = Ложь;
		Страницы["ЧисловойДиапазон"].Видимость = Истина;
		
		ЭлементыФормы.НадписьМинЗначение .Заголовок = "Значение:";
		ЭлементыФормы.НадписьМаксЗначение.Заголовок = "Погрешность (+/-):";
		
		ЭлементыФормы.ПанельДопустимыеЗначениеПоказателей.ТекущаяСтраница = Страницы["ЧисловойДиапазон"];
		
		ЭлементыФормы.НадписьТекстПояснение.Заголовок = "Допустимым результатом замеров будет число входящее в указанные пределы (число +/- погрешность).";
		
	ИначеЕсли ВидРезультатаАнализа = ПВХ.Поддиапазон  Тогда
		
		Страницы["ВхождениеВСписок"].Видимость = Ложь;
		Страницы["ЧисловойДиапазон"].Видимость = Истина;
		
		ЭлементыФормы.НадписьМинЗначение .Заголовок = "Начало диапазона:";
		ЭлементыФормы.НадписьМаксЗначение.Заголовок = "Конец диапазона:";
		
		ЭлементыФормы.ПанельДопустимыеЗначениеПоказателей.ТекущаяСтраница = Страницы["ЧисловойДиапазон"];
		
		ЭлементыФормы.НадписьТекстПояснение.Заголовок = "Допустимым результатом замеров будет числовой диапазон лежащий внутри указанного диапазона значений.";
		
	КонецЕсли;
	
КонецПроцедуры // УстановитьВидимость()

Процедура ВидРезультатаАнализаПриИзменении(Элемент)
	
	УстановитьВидимость();
	
КонецПроцедуры // ВидРезультатаАнализаПриИзменении()

Процедура ПриОткрытии()
	
	УстановитьВидимость();
	
	МеханизмНумерацииОбъектов.ДобавитьВМенюДействияКнопкуРедактированияКода(ЭлементыФормы.ДействияФормы.Кнопки.Подменю);
	МеханизмНумерацииОбъектов.УстановитьДоступностьПоляВводаНомера(Метаданные(), ЭтаФорма, ЭлементыФормы.ДействияФормы.Кнопки.Подменю,ЭлементыФормы.Код);
	
КонецПроцедуры // ПриОткрытии()

Процедура ПослеЗаписи()
	
	МеханизмНумерацииОбъектов.ОбновитьПодсказкуКодНомерОбъекта(ЭтотОбъект.Метаданные(), ЭлементыФормы.ДействияФормы.Кнопки.Подменю, ЭлементыФормы.Код);
	
КонецПроцедуры
