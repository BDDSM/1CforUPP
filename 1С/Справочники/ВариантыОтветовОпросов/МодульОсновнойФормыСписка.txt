Процедура ПередОткрытием(Отказ, СтандартнаяОбработка)
	
	Если ЭтаФорма.ВладелецФормы = Неопределено Тогда
		РаботаСДиалогами.ВывестиПредупреждение("Справочник предназначен для использования при редактировании вопросов типовых анкет и не предназначен для открытия из пункта ""Операции"" стандартного меню!");
		Отказ = Истина;
	КонецЕсли;
	
КонецПроцедуры

Процедура ОбновлениеОтображения()
	
	Если ПараметрОтборПоВладельцу <> Неопределено Тогда
		Заголовок = "Варианты ответов на вопрос: " + ПараметрОтборПоВладельцу.Наименование;
		
	Иначе
		Заголовок = "Варианты ответов опросов";
		
	КонецЕсли;
	
КонецПроцедуры // ОбновлениеОтображения()
